function bind(func, context) {
  return function() { 
    return func.apply(context, arguments); 
  };
}

function User() {
  this.id = 1; 

  this.sayHi = bind(function() { 
    alert(this.id); 
  }, this);
}
var user = new User();
setTimeout(user.sayHi, 1000); // выведет "1"
