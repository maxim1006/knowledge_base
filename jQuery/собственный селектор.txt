$.extend($.expr[':'], {  
    over100pixels: function(a) {  
        return $(a).height() > 100;  
    }  
});  
  
$('.box:over100pixels').click(function() {  
    alert('Элемент, который вы потревожили своим кликом, в высоту более 100 пикселов');
});  


$.extend($.expr[':'], {
    toowide: function(e) {
        return $(e).width() > 640;
    }
});
 
$('.box:toowide').hide();
